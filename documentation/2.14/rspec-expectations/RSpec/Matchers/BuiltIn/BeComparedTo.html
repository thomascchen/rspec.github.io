def failure_message_for_should_not
  message = <<-MESSAGE

it is a bit confusing.
  MESSAGE

  raise message << ([:===,:==].include?(@operator) ?
                    "It might be more clearly expressed without the \"be\"?" :
                    "It might be more clearly expressed in the positive?")
end
