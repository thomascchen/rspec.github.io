def __delegate_operator(actual, operator, expected)
  return false unless actual.__send__(operator, expected)
  return fail_with_message("expected not: #{operator} #{expected.inspect}\n         got: #{operator.gsub(/./, ' ')} #{actual.inspect}")
end
