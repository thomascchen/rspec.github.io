def find_all(*args)
  original_path_set.find_all(*args).collect do |template|
    ::ActionView::Template.new(
      "",
      template.identifier,
      EmptyTemplateHandler,
      {
        :virtual_path => template.virtual_path,
        :format => template.formats
      }
    )
  end
end
