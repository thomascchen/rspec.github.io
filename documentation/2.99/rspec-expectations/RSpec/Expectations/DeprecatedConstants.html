def const_missing(name)
  case name
  when :Rspec, :Spec
    RSpec.deprecate(name.to_s, :replacement => "RSpec")
    RSpec
  else
    begin
      super
    rescue Exception => e
      e.backtrace.reject! {|l| l =~ Regexp.compile(__FILE__) }
      raise e
    end
  end
end
